\section{Cài đặt}


\begin{frame}[fragile]{Code DFS kiểm tra liên thông}
\begin{tcolorbox}
\begin{lstlisting}
vector<vector<int>> adj;
vector<bool> da_tham;

void dfs_lt(int u) {
    da_tham[u] = true;
    for (int v : adj[u]) {
        if (!da_tham[v]) dfs_lt(v);
    }
}

bool kt_lt(int n) {
    da_tham.assign(n, false);
    dfs_lt(0);

    for (int i = 0; i < n; i++) {
        if (!da_tham[i]) return false;
    }
    return true;
}
\end{lstlisting}
\end{tcolorbox}
\end{frame}

\begin{frame}[fragile]{Đếm thành phần liên thông}
\begin{tcolorbox}
\begin{lstlisting}
int dem_tplt(int n) {
    da_tham.assign(n, false);
    int dem = 0;

    for (int i = 0; i < n; i++) {
        if (!da_tham[i]) {
            dfs_lt(i);
            dem++;
        }
    }
    return dem;
}
\end{lstlisting}
\end{tcolorbox}
\end{frame}
